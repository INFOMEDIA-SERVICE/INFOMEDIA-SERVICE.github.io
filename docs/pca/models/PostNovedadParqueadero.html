PostNovedadParqueadero

  • idTipoNovedad : Long
    id del tipo de novedad
  • descripcion : String
    descripcion de la novedad
  • fechaHoraReporte : String
    fecha y hora de reporte de la novedad
  • placa : String
    placa del vehiculo implicado en la novedad
  • idReserva : UUID
    id de la reserva implicada en la novedad
  • tipoIdentificacion : String
    tipo de identificacion de quien reporta
    Enum:
    CC
    NIT
    CE
    PAS
    TI
    DNI
    PEP
    RC
  • numeroIdentificacion : String
    numero de identificacion de quien reporta
  • nombre : String
    nombre de quien reporta
  • propietario : Boolean
    quien reporta es el propietario (si/no)
  • telefono : String
    telefono de quien reporta
  • email : String
    direccion de correo electronico de quien reporta
