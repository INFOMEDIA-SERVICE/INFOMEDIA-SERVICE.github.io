MetodoPago

  • fechaCreado : String
  • creadoPor : String
  • fechaModificado : String
  • modificadoPor : String
  • id : Long
    id del metodo de pago
  • nombre : String
    nombre del metodo de pago
  • cuentaDestino : String
    cuenta destino a la cual se transfieren los pago realizados
  • tipo : String
    tipo de metodo de pago
    Enum:
    EFECTIVO
    ELECTRONICO
    DATAFONO
  • requiereDatosAutorizacion : Boolean
    el metodo de pago requiere datos de autorizacion (si/no)
  • recepcionesPago : List
  • estado : String
    estado del metodo de pago
    Enum:
    ACTIVO
    INACTIVO
