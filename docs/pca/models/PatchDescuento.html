PatchDescuento

  • idDescuento : UUID
  • nombre : String
    nombre del descuento
  • tipo : String
    tipo de descuento
    Enum:
    PORCENTAJE
    MONTO_FIJO
  • aplicarConCodigo : Boolean
    determina si el descuento se aplica usando el codigo, de lo contrario se aplicara a todas las compras de productos que cumplan con las condiciones
  • codigo : String
    codigo del descuento
  • aplicarConVigencia : Boolean
    determina si el descuento se aplica usando vigencia de tiempo, de lo contrario se aplicara a todas las compras de productos sin importar la fecha
  • vigenciaInicio : String
    fecha de inicio de vigencia del descuento
  • vigenciaFin : String
    fecha de finalizacion de vigencia del descuento
  • tiposProducto : List
  • valor : Double
    valor del descuento
  • consumoMax : Integer
    numero de consumos maximo del descuento, 0 para consumo ilimitado
  • consumoMaxCliente : Integer
    numero de consumos maximo por cliente, 0 para consumo ilimitado
  • clientesObjetivo : Boolean
    determina si el descuento se aplica a los clientes objetivo, de lo contrario se aplicara a todos los clientes
  • clientesCompraronEnFechas : Boolean
    determina si los clientes objetivo seran los que hayan comprado dentro de el rango de fechas especificado (si/no)
  • ccefFechaInicio : String
    fecha de inicio en la que los clientes objetivo compraron
  • ccefFechaFin : String
    fecha de finalizacion en la que los clientes objetivo compraron
  • clientesEspecificos : Boolean
    determina si los clientes objetivo seran clientes especificos (si/no)
  • ceExtension : String
    extension del archivo plano para la carga de clientes especificos
    Enum:
    CSV
    XLSX
  • ceBase64Data : String
    datos binarios del archivo plano para carga de clientes especificos en base64
